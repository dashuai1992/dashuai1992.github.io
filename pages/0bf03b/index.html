使用docker-compose安装mysql

# docker-compose.yaml

services:
  mysql:
    restart: always
    image: mysql
    ports:
      - 3306:3306
    command:
      --character-set-server=utf8mb4
      --collation-server=utf8mb4_general_ci
    environment:
      MYSQL_ROOT_PASSWORD: root
    volumes:
      - ./log:/var/log/mysql
      - ./data:/var/lib/mysql
      - ./cnf:/etc/mysql/conf.d
      - /etc/localtime:/etc/localtime
